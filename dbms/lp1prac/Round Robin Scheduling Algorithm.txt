#include<iostream>
using namespace std;
int main(){
	int    at[20],bt[20],ct[20],cpu[20],tat[20],wt[20],rt[20];
	
	int n,j=1,s=0;
	float avgtat,avgwt;
	cout<<"Enter the number of process: ";
	cin>>n;
	for(int i=0;i<n;i++){
	    cout<<"Enter the arrival time for process "<<i+1<<": ";
		cin>>at[i];
	}
	cout<<"\n";
	
	for(int i=0;i<n;i++){
	    cout<<"Enter the burst time process"<<i+1<<": ";
		cin>>bt[i];
	}
	cout<<"\n";
	
	for(int i=0;i<n;i++){
	    cout<<"Enter the copmletion time process"<<i+1<<": ";
		cin>>ct[i];
	}
	cout<<"\n";
	
	for(int i=0;i<n;i++){
	    cout<<"Enter the CPU first time process"<<i+1<<": ";
		cin>>cpu[i];
	}
	cout<<"\n";
	

	for(int i=0;i<n;i++){
		tat[i]=ct[i]-at[i];
	}

	for(int i=0;i<n;i++){
		wt[i]=tat[i]-bt[i];
	}

	for(int i=0;i<n;i++){
		rt[i]=cpu[i]-at[i];
	}
	cout<<"Tabular format for RR is - \n";
	
    cout<<"| Process Number |    |  Arrival Time  |    |  Burst time  |    |  Turnaround time  |    |  Waiting time |   Response Time";     
	for(int i=0;i<n;i++){
		cout<<"\n       P"<<j++<<"\t\t\t"<<at[i]<<"\t\t    "<<bt[i]<<"\t\t       "<<tat[i]<<"\t\t\t    "<<wt[i]<<"\t\t\t"  <<rt[i];
		cout<<"\n";
	}
	cout<<"________________________________________________________________________________________________________";
	cout<<"\n\n";
	cout<<"Average turnaround time is : ";
	for(int i=0;i<n;i++){
		avgtat = avgtat + tat[i];
	}
	cout<<avgtat/n;
	cout<<"\n";
	cout<<"Average waiting time is : ";
	for(int i=0;i<n;i++)
	{
		avgwt = avgwt + wt[i];
	}
	cout<<avgwt/n<<"\n";
	cout<<"\n\n";
	
}
